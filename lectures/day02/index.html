Getters, Setters & Access Modifiers

					

					

						
Tesla Class

						

							// Tesla.java
							class Tesla {
								private String owner = "Tyler";
								private String color = "red";
								private int Doors = 4;
							}
						

					

					

						
Tesla Class

						

							// Tesla.java
							class Tesla {
								private String owner = "Tyler";
								private String color = "red";
								private int Doors = 4;

								// getters and setters
								public String getOwner() {
									return owner;
								}
							
								public void setOwner(String owner) {
									this.owner = owner;
								}
							
								public String getColor() {
									return color;
								}
							
								public void setColor(String color) {
									this.color = color;
								}
							
								public int getDoors() {
									return Doors;
								}
							
								public void setDoors(int doors) {
									Doors = doors;
								}
							}
