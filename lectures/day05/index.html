Java Spring

@RestController and Annotations

HomeController.java


						package com.codingdojo.controllerspractice.controllers;
						import org.springframework.web.bind.annotation.RequestMapping;
						import org.springframework.web.bind.annotation.RestController;
						@RestController
						public class HomeController {
								@RequestMapping("/")
								public String hello() {
										return "Hello World!";
								}
						}

@RestController - allows our controller to respond with data (ie. strings, json)

@RequestMapping - maps web requests onto classes and/or methods
